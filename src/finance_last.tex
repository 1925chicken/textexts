    $
    \left(
    \begin{array}{cc}
        0.5 & 1 \\
        1   & 3
    \end{array}
    \right)
    = \left(\begin{array}{cc}
                0.5 & 1 \\
                0   & 1
    \end{array}
    \right)\\
    = \left(
    \begin{array}{cc}
        0.5 & 0 \\
        0   & 1
    \end{array}
    \right)
    = \left(
    \begin{array}{cc}
        1 & 0 \\
        0 & 1
    \end{array}
    \right)
    $\\
